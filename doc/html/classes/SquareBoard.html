  def subclass_init
    @name = "Square Board (#{@side}x#{@side})"
    @expansion = StandardGame.new
    @recomended_players = 2..4
  
    bag = TileBag.new(@side ** 2)
    #Tiles

    coords = []
    for x in 1..@side do
      for y in 1..@side do
        coords << [x,y]
      end
    end
    for c in coords
      @tiles[c] = @tile_bag.grab
      @tiles[c].coords = c
    end
    connectTiles
  end
