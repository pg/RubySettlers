  def get_adjecent_edges
    result = []
    for n in @nodes
      result += n.edges
    end
    result.delete(self)
    result.uniq
  end

  def visit_road(visitedEdges=[])
    if @road
      visitedEdges << self
      yield self
      for e in get_adjecent_edges
        if not visitedEdges.include?(e) 
          if e.road and e.road.color == @road.color
            e.visit_road(visitedEdges){|edge| yield edge}
          end
        end
      end
    end
  end
