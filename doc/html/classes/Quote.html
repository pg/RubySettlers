  def Quote.from_json(hash, connection)
    tradee = nil
    if hash['tradee']
      tradee = PlayerInfo.from_json(hash['tradee'], connection)
    end
    receiveType = Kernel.const_get(hash['receiveType'].to_sym)
    receiveNum = hash['receiveNum']
    giveType = Kernel.const_get(hash['giveType'].to_sym)
    giveNum = hash['giveNum']
    Quote.new(tradee, receiveType,receiveNum, giveType, giveNum) 
  end

  def initialize(bidder, receiveType, receiveNum, giveType, giveNum)
